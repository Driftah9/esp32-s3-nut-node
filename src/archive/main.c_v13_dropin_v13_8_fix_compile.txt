 *
 * NOTE: This is a drop-in single-file build for ESP-IDF v5.3.x.
 */

static const char *TAG = "UPS_USB_M3";

/* =========================
 * Configuration (NVS)
 * ========================= */
#define NVS_NS "cfg"

typedef struct {
    char sta_ssid[33];
    char sta_pass[65];

    char ap_pass[65];     // if empty => open AP
    char nut_user[33];    // basic auth for upsd-like interface
    char nut_pass[65];

    char ups_name[33];    // e.g. "ups"
} app_cfg_t;

static app_cfg_t g_cfg;


// Runtime UPS data exposed via upsd (populated by future USB/HID decode tasks)
typedef struct {
    int battery_charge;            // percent
    int battery_runtime;           // seconds
    int input_utility_present;     // 0/1
    uint32_t ups_flags;            // bitmask
    char ups_status[16];           // e.g. "OL", "OB", etc.
} ups_runtime_t;

static ups_runtime_t g_ups = {
    .battery_charge = 0,
    .battery_runtime = 0,
    .input_utility_present = 0,
    .ups_flags = 0,
    .ups_status = "INIT",
};

static void cfg_defaults(app_cfg_t *c) {
    memset(c, 0, sizeof(*c));
    // No STA creds by default (required by user).
    // SoftAP password default empty => open AP until user sets.
    strncpy(c->nut_user, "admin", sizeof(c->nut_user)-1);
    strncpy(c->nut_pass, "admin", sizeof(c->nut_pass)-1);
    strncpy(c->ups_name, "ups", sizeof(c->ups_name)-1);
}

static esp_err_t nvs_get_str_safe(nvs_handle_t h, const char *key, char *out, size_t out_sz) {
    size_t req = 0;
    esp_err_t err = nvs_get_str(h, key, NULL, &req);
    if (err == ESP_ERR_NVS_NOT_FOUND) return err;
    if (err != ESP_OK) return err;
    if (req == 0) return ESP_ERR_INVALID_SIZE;
    if (req > out_sz) return ESP_ERR_INVALID_SIZE;
    return nvs_get_str(h, key, out, &req);
}

static void cfg_load(app_cfg_t *c) {
    cfg_defaults(c);

    nvs_handle_t h;
    esp_err_t err = nvs_open(NVS_NS, NVS_READONLY, &h);
    if (err != ESP_OK) {
        ESP_LOGW(TAG, "NVS cfg not found yet (first boot).");
        return;
    }

    (void)nvs_get_str_safe(h, "sta_ssid", c->sta_ssid, sizeof(c->sta_ssid));
    (void)nvs_get_str_safe(h, "sta_pass", c->sta_pass, sizeof(c->sta_pass));
    (void)nvs_get_str_safe(h, "ap_pass",  c->ap_pass,  sizeof(c->ap_pass));
    (void)nvs_get_str_safe(h, "nut_user", c->nut_user, sizeof(c->nut_user));
    (void)nvs_get_str_safe(h, "nut_pass", c->nut_pass, sizeof(c->nut_pass));
    (void)nvs_get_str_safe(h, "ups_name", c->ups_name, sizeof(c->ups_name));

    nvs_close(h);
}

static esp_err_t cfg_save(const app_cfg_t *c) {
    nvs_handle_t h;
    esp_err_t err = nvs_open(NVS_NS, NVS_READWRITE, &h);
    if (err != ESP_OK) return err;

    // store even if empty (clears keys)
    err = nvs_set_str(h, "sta_ssid", c->sta_ssid); if (err != ESP_OK) goto out;
    err = nvs_set_str(h, "sta_pass", c->sta_pass); if (err != ESP_OK) goto out;
    err = nvs_set_str(h, "ap_pass",  c->ap_pass);  if (err != ESP_OK) goto out;
    err = nvs_set_str(h, "nut_user", c->nut_user); if (err != ESP_OK) goto out;
    err = nvs_set_str(h, "nut_pass", c->nut_pass); if (err != ESP_OK) goto out;
    err = nvs_set_str(h, "ups_name", c->ups_name); if (err != ESP_OK) goto out;

    err = nvs_commit(h);
out:
    nvs_close(h);
    return err;
}

/* =========================
 * Wi-Fi (APSTA + config portal)
 * ========================= */
static EventGroupHandle_t wifi_event_group;
#define WIFI_STA_GOT_IP_BIT BIT0

static esp_netif_t *s_netif_sta = NULL;
static esp_netif_t *s_netif_ap  = NULL;

static void make_softap_ssid(char *out, size_t out_sz) {
    uint8_t mac[6] = {0};
    esp_read_mac(mac, ESP_MAC_WIFI_SOFTAP);
    // e.g. ESP32-UPS-SETUP-4AE49D
    snprintf(out, out_sz, "ESP32-UPS-SETUP-%02X%02X%02X", mac[3], mac[4], mac[5]);
}

static void wifi_start_apsta(const app_cfg_t *cfg);

static void on_wifi_event(void *arg, esp_event_base_t base, int32_t id, void *data) {
    if (base == WIFI_EVENT) {
        if (id == WIFI_EVENT_STA_START) {
            if (g_cfg.sta_ssid[0]) {
                ESP_LOGI(TAG, "STA start: connecting to '%s'...", g_cfg.sta_ssid);
                esp_wifi_connect();
            } else {
                ESP_LOGI(TAG, "STA start: no STA creds yet (configure via SoftAP).");
            }
        } else if (id == WIFI_EVENT_STA_DISCONNECTED) {
            xEventGroupClearBits(wifi_event_group, WIFI_STA_GOT_IP_BIT);
            ESP_LOGW(TAG, "STA disconnected.");
        } else if (id == WIFI_EVENT_AP_START) {
            ESP_LOGI(TAG, "SoftAP started (config portal).");
        }
    }
    (void)arg; (void)data;
}

// NOTE: We can't capture cfg in event handler directly via arg in this single-file drop-in easily,
// so we'll use global g_cfg for now.
static void wifi_event_router(void *arg, esp_event_base_t base, int32_t id, void *data) {
    (void)arg;
    if (base == WIFI_EVENT && id == WIFI_EVENT_STA_START) {
        if (g_cfg.sta_ssid[0]) {
            ESP_LOGI(TAG, "STA start: connecting to '%s'...", g_cfg.sta_ssid);
            esp_wifi_connect();
        } else {
            ESP_LOGI(TAG, "STA start: no STA creds yet (configure via SoftAP).");
        }
        return;
    }
    if (base == WIFI_EVENT && id == WIFI_EVENT_STA_DISCONNECTED) {
        xEventGroupClearBits(wifi_event_group, WIFI_STA_GOT_IP_BIT);
        ESP_LOGW(TAG, "STA disconnected.");
        return;
    }
    if (base == WIFI_EVENT && id == WIFI_EVENT_AP_START) {
        ESP_LOGI(TAG, "SoftAP started (config portal).");
        return;
    }
}

static void ip_event_router(void *arg, esp_event_base_t base, int32_t id, void *data) {
    (void)arg;
    if (base == IP_EVENT && id == IP_EVENT_STA_GOT_IP) {
        ip_event_got_ip_t *ev = (ip_event_got_ip_t *)data;
        ESP_LOGI(TAG, "STA got IP: " IPSTR, IP2STR(&ev->ip_info.ip));
        xEventGroupSetBits(wifi_event_group, WIFI_STA_GOT_IP_BIT);
    }
}

static void wifi_start_apsta(const app_cfg_t *cfg) {
    wifi_event_group = xEventGroupCreate();

    ESP_ERROR_CHECK(esp_netif_init());
    ESP_ERROR_CHECK(esp_event_loop_create_default());

    s_netif_sta = esp_netif_create_default_wifi_sta();
    s_netif_ap  = esp_netif_create_default_wifi_ap();

    wifi_init_config_t wcfg = WIFI_INIT_CONFIG_DEFAULT();
    ESP_ERROR_CHECK(esp_wifi_init(&wcfg));

    ESP_ERROR_CHECK(esp_event_handler_register(WIFI_EVENT, ESP_EVENT_ANY_ID, &wifi_event_router, NULL));
    ESP_ERROR_CHECK(esp_event_handler_register(IP_EVENT, IP_EVENT_STA_GOT_IP, &ip_event_router, NULL));

    ESP_ERROR_CHECK(esp_wifi_set_storage(WIFI_STORAGE_RAM));
    ESP_ERROR_CHECK(esp_wifi_set_mode(WIFI_MODE_APSTA));

    // SoftAP config
    char ap_ssid[32];
    make_softap_ssid(ap_ssid, sizeof(ap_ssid));

    wifi_config_t ap_cfg = {0};
    strncpy((char *)ap_cfg.ap.ssid, ap_ssid, sizeof(ap_cfg.ap.ssid)-1);
    ap_cfg.ap.ssid_len = (uint8_t)strlen(ap_ssid);
    ap_cfg.ap.max_connection = 4;
    ap_cfg.ap.channel = 6;

    if (cfg->ap_pass[0]) {
        strncpy((char *)ap_cfg.ap.password, cfg->ap_pass, sizeof(ap_cfg.ap.password)-1);
        ap_cfg.ap.authmode = WIFI_AUTH_WPA2_PSK;
    } else {
        ap_cfg.ap.password[0] = '\0';
        ap_cfg.ap.authmode = WIFI_AUTH_OPEN;
    }

    ESP_ERROR_CHECK(esp_wifi_set_config(WIFI_IF_AP, &ap_cfg));

    // STA config (only if present)
    if (cfg->sta_ssid[0]) {
        wifi_config_t sta_cfg = {0};
        strncpy((char *)sta_cfg.sta.ssid, cfg->sta_ssid, sizeof(sta_cfg.sta.ssid)-1);
        strncpy((char *)sta_cfg.sta.password, cfg->sta_pass, sizeof(sta_cfg.sta.password)-1);
        sta_cfg.sta.threshold.authmode = WIFI_AUTH_WPA2_PSK;
        sta_cfg.sta.pmf_cfg.capable = true;
        sta_cfg.sta.pmf_cfg.required = false;
        ESP_ERROR_CHECK(esp_wifi_set_config(WIFI_IF_STA, &sta_cfg));
    }

    ESP_ERROR_CHECK(esp_wifi_start());

    ESP_LOGI(TAG, "SoftAP ready: connect to SSID '%s' then open http://192.168.4.1/config", ap_ssid);
}

/* =========================
 * Minimal URL decode + form parsing
 * ========================= */
static int hexval(char c) {
    if (c >= '0' && c <= '9') return c - '0';
    if (c >= 'a' && c <= 'f') return 10 + (c - 'a');
    if (c >= 'A' && c <= 'F') return 10 + (c - 'A');
    return -1;
}

static void urldecode_inplace(char *s) {
    char *o = s;
    for (char *p = s; *p; p++) {
        if (*p == '+') { *o++ = ' '; }
        else if (*p == '%' && p[1] && p[2]) {
            int h1 = hexval(p[1]), h2 = hexval(p[2]);
            if (h1 >= 0 && h2 >= 0) {
                *o++ = (char)((h1 << 4) | h2);
                p += 2;
            } else {
                *o++ = *p;
            }
        } else {
            *o++ = *p;
        }
    }
    *o = '\0';
}

static void form_get(const char *body, const char *key, char *out, size_t out_sz) {
    out[0] = '\0';
    if (!body || !key) return;

    size_t klen = strlen(key);
    const char *p = body;
    while (*p) {
        const char *amp = strchr(p, '&');
        size_t seglen = amp ? (size_t)(amp - p) : strlen(p);

        const char *eq = memchr(p, '=', seglen);
        if (eq) {
            size_t nlen = (size_t)(eq - p);
            if (nlen == klen && strncmp(p, key, klen) == 0) {
                size_t vlen = seglen - (nlen + 1);
                if (vlen >= out_sz) vlen = out_sz - 1;
                memcpy(out, eq + 1, vlen);
                out[vlen] = '\0';
                urldecode_inplace(out);
                return;
            }
        }
        if (!amp) break;
        p = amp + 1;
    }
}

/* =========================

static httpd_handle_t s_httpd = NULL;

static esp_err_t send_html(httpd_req_t *req, const char *html) {
    httpd_resp_set_type(req, "text/html");
    return httpd_resp_send(req, html, HTTPD_RESP_USE_STRLEN);
}

static esp_err_t status_get(httpd_req_t *req) {
    char sta_ip[64] = "N/A";
    esp_netif_ip_info_t ip;
    if (s_netif_sta && esp_netif_get_ip_info(s_netif_sta, &ip) == ESP_OK && ip.ip.addr != 0) {
        snprintf(sta_ip, sizeof(sta_ip), IPSTR, IP2STR(&ip.ip));
    }

    char buf[1024];
    snprintf(buf, sizeof(buf),
        "<!doctype html><html><head><meta charset='utf-8'><title>ESP32 UPS</title></head><body>"
        "<h2>ESP32 UPS NUT Node</h2>"
        "<p><b>SoftAP:</b> 192.168.4.1</p>"
        "<p><b>STA IP:</b> %s</p>"
        "<p><a href='/config'>Configure Wi-Fi / NUT</a></p>"
        "<p>NUT TCP port: <b>3493</b></p>"
        "</body></html>",
        sta_ip
    );
    return send_html(req, buf);
}

static bool try_sta_connect_blocking(const app_cfg_t *cfg, int timeout_ms) {
    if (!cfg->sta_ssid[0]) return false;

    // apply config and connect
    wifi_config_t sta_cfg = {0};
    strncpy((char *)sta_cfg.sta.ssid, cfg->sta_ssid, sizeof(sta_cfg.sta.ssid)-1);
    strncpy((char *)sta_cfg.sta.password, cfg->sta_pass, sizeof(sta_cfg.sta.password)-1);
    sta_cfg.sta.threshold.authmode = WIFI_AUTH_WPA2_PSK;
    sta_cfg.sta.pmf_cfg.capable = true;
    sta_cfg.sta.pmf_cfg.required = false;
    ESP_ERROR_CHECK(esp_wifi_set_config(WIFI_IF_STA, &sta_cfg));

    xEventGroupClearBits(wifi_event_group, WIFI_STA_GOT_IP_BIT);
    esp_wifi_disconnect();
    esp_wifi_connect();

    EventBits_t bits = xEventGroupWaitBits(wifi_event_group, WIFI_STA_GOT_IP_BIT, pdFALSE, pdTRUE,
                                          pdMS_TO_TICKS(timeout_ms));
    return (bits & WIFI_STA_GOT_IP_BIT) != 0;
}

static esp_err_t save_post(httpd_req_t *req) {
    int len = req->content_len;
    if (len <= 0 || len > 2048) {
        httpd_resp_send_err(req, HTTPD_400_BAD_REQUEST, "Bad form length");
        return ESP_OK;
    }

    char *body = (char *)calloc(1, len + 1);
    if (!body) {
        httpd_resp_send_err(req, HTTPD_500_INTERNAL_SERVER_ERROR, "OOM");
        return ESP_OK;
    }

    int r = httpd_req_recv(req, body, len);
    if (r <= 0) {
        free(body);
        httpd_resp_send_err(req, HTTPD_500_INTERNAL_SERVER_ERROR, "recv failed");
        return ESP_OK;
    }
    body[r] = '\0';

    app_cfg_t newc = g_cfg;

    form_get(body, "sta_ssid", newc.sta_ssid, sizeof(newc.sta_ssid));
    form_get(body, "sta_pass", newc.sta_pass, sizeof(newc.sta_pass));
    form_get(body, "ap_pass",  newc.ap_pass,  sizeof(newc.ap_pass));
    form_get(body, "ups_name", newc.ups_name, sizeof(newc.ups_name));
    form_get(body, "nut_user", newc.nut_user, sizeof(newc.nut_user));

    char nut_pass_in[65];
    form_get(body, "nut_pass", nut_pass_in, sizeof(nut_pass_in));
    if (nut_pass_in[0]) {
        strncpy(newc.nut_pass, nut_pass_in, sizeof(newc.nut_pass)-1);
    }

    free(body);

    // Save to NVS first (so reboot keeps it)
    esp_err_t err = cfg_save(&newc);
    if (err != ESP_OK) {
        httpd_resp_send_err(req, HTTPD_500_INTERNAL_SERVER_ERROR, "NVS save failed");
        return ESP_OK;
    }

    // Update live config in RAM
    g_cfg = newc;

    bool ok = false;
    if (g_cfg.sta_ssid[0]) {
        ok = try_sta_connect_blocking(&g_cfg, 12000);
    }

    char resp[1024];
    if (g_cfg.sta_ssid[0] && ok) {
        esp_netif_ip_info_t ip;
        char ipbuf[64] = "N/A";
        if (s_netif_sta && esp_netif_get_ip_info(s_netif_sta, &ip) == ESP_OK && ip.ip.addr != 0) {
            snprintf(ipbuf, sizeof(ipbuf), IPSTR, IP2STR(&ip.ip));
        }
        snprintf(resp, sizeof(resp),
            "<b>Saved.</b> <b>STA connect OK.</b> IP: %s &nbsp; "
            "<a href=\"/status\">Status</a> | <a href=\"/config\">Back</a> | "
            "<a href=\"/reboot\">Reboot</a>", ipbuf);
        send_html(req, resp);
    } else if (g_cfg.sta_ssid[0] && !ok) {
        snprintf(resp, sizeof(resp),
            "<b>Saved.</b> <span style=\"color:#c00\"><b>STA connect FAILED.</b></span> "
            "SoftAP still at <b>192.168.4.1</b>. &nbsp; "
            "<a href=\"/config\">Back</a>");
        send_html(req, resp);
    } else {
        snprintf(resp, sizeof(resp),
            "<b>Saved.</b> STA disabled (SSID empty). SoftAP-only. &nbsp; "
            "<a href=\"/status\">Status</a> | <a href=\"/config\">Back</a>");
        send_html(req, resp);
    }

    return ESP_OK;
}

static esp_err_t reboot_post(httpd_req_t *req) {
    httpd_resp_sendstr(req, "Rebooting...\n");
    vTaskDelay(pdMS_TO_TICKS(200));
    esp_restart();
    return ESP_OK;
}

static esp_err_t reboot_get(httpd_req_t *req) {
    return reboot_post(req);
}

static void start_http_server(void) {
    httpd_config_t config = HTTPD_DEFAULT_CONFIG();
    config.stack_size = 8192;
    // NOTE: In ESP-IDF v5.3.x, request-header/URI length limits are set via sdkconfig
    // (CONFIG_HTTPD_MAX_REQ_HDR_LEN / CONFIG_HTTPD_MAX_URI_LEN), not via httpd_config_t.
    // If you see 431 "Request Header Fields Too Large" when posting /save from a browser,
    // raise CONFIG_HTTPD_MAX_REQ_HDR_LEN (suggest 2048) and CONFIG_HTTPD_MAX_URI_LEN (suggest 512).
#if defined(CONFIG_HTTPD_MAX_REQ_HDR_LEN) && (CONFIG_HTTPD_MAX_REQ_HDR_LEN < 2048)
    ESP_LOGW(TAG, "CONFIG_HTTPD_MAX_REQ_HDR_LEN=%d may be too small for modern browsers; consider 2048.", CONFIG_HTTPD_MAX_REQ_HDR_LEN);
#endif
#if defined(CONFIG_HTTPD_MAX_URI_LEN) && (CONFIG_HTTPD_MAX_URI_LEN < 512)
    ESP_LOGW(TAG, "CONFIG_HTTPD_MAX_URI_LEN=%d may be too small; consider 512.", CONFIG_HTTPD_MAX_URI_LEN);
#endif

    ESP_ERROR_CHECK(httpd_start(&s_httpd, &config));

    httpd_uri_t u_status = {.uri="/status", .method=HTTP_GET, .handler=status_get, .user_ctx=NULL};
    httpd_uri_t u_config = {.uri="/config", .method=HTTP_GET, .handler=config_get, .user_ctx=NULL};
    httpd_uri_t u_save   = {.uri="/save",   .method=HTTP_POST, .handler=save_post, .user_ctx=NULL};
    httpd_uri_t u_reboot = {.uri="/reboot", .method=HTTP_POST, .handler=reboot_post, .user_ctx=NULL};
    httpd_uri_t u_reboot_get = {.uri="/reboot", .method=HTTP_GET, .handler=reboot_get, .user_ctx=NULL};

    ESP_ERROR_CHECK(httpd_register_uri_handler(s_httpd, &u_status));
    ESP_ERROR_CHECK(httpd_register_uri_handler(s_httpd, &u_config));
    ESP_ERROR_CHECK(httpd_register_uri_handler(s_httpd, &u_save));
    ESP_ERROR_CHECK(httpd_register_uri_handler(s_httpd, &u_reboot));
    ESP_ERROR_CHECK(httpd_register_uri_handler(s_httpd, &u_reboot_get));

    ESP_LOGI(TAG, "HTTP server started: GET http://192.168.4.1/status");
}

/* =========================
 * NUT-ish TCP server (upsd-like)
 * ========================= */
static int s_nut_listen_fd = -1;

static void nut_send(int fd, const char *s) {
    if (!s) return;
    send(fd, s, (int)strlen(s), 0);
}

static void nut_sendf(int fd, const char *fmt, ...) {
    char buf[256];
    va_list ap;
    va_start(ap, fmt);
    vsnprintf(buf, sizeof(buf), fmt, ap);
    va_end(ap);
    nut_send(fd, buf);
}

static void nut_handle_client(int fd)
{
    // Minimal NUT upsd-like protocol handler for tcp/3493.
    // Important: upsd does NOT send an unsolicited banner on connect; clients issue VER/NETVER/etc.
    // Also: clients may pipeline multiple commands in one TCP segment, so we must parse line-by-line.

    char rxbuf[1024];
    size_t used = 0;

    bool authed = false;
    char pending_user[64] = {0};

    // Helper lambdas (C doesn't have them): small local functions via macros.
#define SENDLN(_fmt, ...) do { \
        char _out[512]; \
        int _n = snprintf(_out, sizeof(_out), _fmt "\n", ##__VA_ARGS__); \
        if (_n > 0) { (void)send(fd, _out, (size_t)_n, 0); } \
    } while (0)

#define TRIM_LINE(_s) do { \
        char *_p = (_s); \
        while (*_p == ' ' || *_p == '\t' || *_p == '\r') _p++; \
        if (_p != (_s)) memmove((_s), _p, strlen(_p) + 1); \
        size_t _l = strlen((_s)); \
        while (_l && (((_s)[_l-1] == '\r') || ((_s)[_l-1] == ' ') || ((_s)[_l-1] == '\t'))) { (_s)[_l-1] = 0; _l--; } \
    } while (0)

    while (1) {
        // Read more data if we don't yet have a full line.
        if (used < sizeof(rxbuf) - 1) {
            int r = recv(fd, rxbuf + used, (int)(sizeof(rxbuf) - 1 - used), 0);
            if (r <= 0) break;
            used += (size_t)r;
            rxbuf[used] = 0;
        }

        // Process complete lines (newline-terminated).
        char *line_start = rxbuf;
        while (1) {
            char *nl = memchr(line_start, '\n', rxbuf + used - line_start);
            if (!nl) break;

            // Extract one line into a separate buffer for parsing.
            size_t linelen = (size_t)(nl - line_start);
            char line[256];
            if (linelen >= sizeof(line)) linelen = sizeof(line) - 1;
            memcpy(line, line_start, linelen);
            line[linelen] = 0;

            // Advance start past newline.
            line_start = nl + 1;

            TRIM_LINE(line);
            if (!line[0]) continue;

            // Uppercase a copy for command matching (preserve original args).
            char uline[256];
            strncpy(uline, line, sizeof(uline));
            uline[sizeof(uline) - 1] = 0;
            for (char *p = uline; *p; p++) if (*p >= 'a' && *p <= 'z') *p = (char)(*p - 32);

            ESP_LOGI(TAG, "NUT<= %s", line);

            // A few NUT clients probe for TLS support; we are plaintext-only for now.
            if (!strcmp(uline, "STARTTLS")) {
                SENDLN("ERR FEATURE-NOT-SUPPORTED");
                continue;
            }

            // Version probes
            if (!strcmp(uline, "VER") || !strcmp(uline, "NETVER")) {
                SENDLN("Network UPS Tools upsd-esp32 0.1");
                continue;
            }

            // Authentication (optional for read-only; we accept and track it for future upsmon use)
            if (!strncmp(uline, "USERNAME ", 9)) {
                strncpy(pending_user, line + 9, sizeof(pending_user) - 1);
                pending_user[sizeof(pending_user) - 1] = 0;
                TRIM_LINE(pending_user);
                SENDLN("OK");
                continue;
            }
            if (!strncmp(uline, "PASSWORD ", 9)) {
                char pass[128];
                strncpy(pass, line + 9, sizeof(pass) - 1);
                pass[sizeof(pass) - 1] = 0;
                TRIM_LINE(pass);

                if (pending_user[0] == 0) {
                    SENDLN("ERR INVALID-ARGUMENT");
                    continue;
                }

                // If configured, enforce creds; otherwise allow any (dev mode).
                if (g_cfg.nut_user[0] && g_cfg.nut_pass[0]) {
                    if (!strcmp(pending_user, g_cfg.nut_user) && !strcmp(pass, g_cfg.nut_pass)) {
                        authed = true;
                        SENDLN("OK");
                    } else {
                        authed = false;
                        SENDLN("ERR ACCESS-DENIED");
                    }
                } else {
                    authed = true;
                    SENDLN("OK");
                }
                continue;
            }

            // LIST UPS
            if (!strcmp(uline, "LIST UPS")) {
                const char *upsname = g_cfg.ups_name[0] ? g_cfg.ups_name : "ups";
                SENDLN("BEGIN LIST UPS");
                SENDLN("UPS %s \"ESP32S3 APC UPS (HID)\"", upsname);
                SENDLN("END LIST UPS");
                continue;
            }

            // LIST VAR <ups>
            if (!strncmp(uline, "LIST VAR ", 9)) {
                const char *upsname = g_cfg.ups_name[0] ? g_cfg.ups_name : "ups";
                // If client asks for a different ups, mimic upsd behavior with UNKNOWN-UPS.
                if (strcmp(line + 9, upsname) != 0) {
                    // Allow both "ups" and configured name for convenience during transition.
                    if (!(strcmp(line + 9, "ups") == 0 && strcmp(upsname, "ups") != 0)) {
                        SENDLN("ERR UNKNOWN-UPS");
                        continue;
                    }
                }
                SENDLN("BEGIN LIST VAR %s", upsname);
                SENDLN("VAR %s battery.charge \"%d\"", upsname, (int)g_ups.battery_charge);
                SENDLN("VAR %s battery.runtime \"%d\"", upsname, (int)g_ups.battery_runtime);
                SENDLN("VAR %s input.utility.present \"%d\"", upsname, g_ups.input_utility_present ? 1 : 0);
                SENDLN("VAR %s ups.flags \"0x%08X\"", upsname, (unsigned)g_ups.flags);
                SENDLN("VAR %s ups.status \"%s\"", upsname, g_ups.status[0] ? g_ups.status : "OL");
                SENDLN("END LIST VAR %s", upsname);
                continue;
            }

            // GET VAR <ups> <var>
            if (!strncmp(uline, "GET VAR ", 8)) {
                char ups[64] = {0}, var[64] = {0};
                (void)sscanf(line + 8, "%63s %63s", ups, var);

                const char *upsname = g_cfg.ups_name[0] ? g_cfg.ups_name : "ups";
                if (ups[0] == 0 || var[0] == 0) {
                    SENDLN("ERR INVALID-ARGUMENT");
                    continue;
                }
                if (strcmp(ups, upsname) != 0 && strcmp(ups, "ups") != 0) {
                    SENDLN("ERR UNKNOWN-UPS");
                    continue;
                }

                if (!strcmp(var, "battery.charge")) {
                    SENDLN("VAR %s battery.charge \"%d\"", upsname, (int)g_ups.battery_charge);
                } else if (!strcmp(var, "battery.runtime")) {
                    SENDLN("VAR %s battery.runtime \"%d\"", upsname, (int)g_ups.battery_runtime);
                } else if (!strcmp(var, "input.utility.present")) {
                    SENDLN("VAR %s input.utility.present \"%d\"", upsname, g_ups.input_utility_present ? 1 : 0);
                } else if (!strcmp(var, "ups.flags")) {
                    SENDLN("VAR %s ups.flags \"0x%08X\"", upsname, (unsigned)g_ups.flags);
                } else if (!strcmp(var, "ups.status")) {
                    SENDLN("VAR %s ups.status \"%s\"", upsname, g_ups.status[0] ? g_ups.status : "OL");
                } else {
                    SENDLN("ERR UNKNOWN-VAR");
                }
                continue;
            }

            // HELP
            if (!strcmp(uline, "HELP")) {
                SENDLN("CMDS: VER NETVER LIST UPS LIST VAR GET VAR USERNAME PASSWORD STARTTLS");
                continue;
            }

            // Default
            (void)authed; // placeholder for future ACLs
            SENDLN("ERR INVALID-COMMAND");
        }

        // Move any remaining partial line to the start of rxbuf.
        size_t remaining = (size_t)(rxbuf + used - line_start);
        if (remaining > 0 && line_start != rxbuf) {
            memmove(rxbuf, line_start, remaining);
        }
        used = remaining;

        // If buffer fills without newline, drop it to avoid deadlock.
        if (used >= sizeof(rxbuf) - 1) {
            used = 0;
        }
    }

#undef SENDLN
#undef TRIM_LINE

    shutdown(fd, SHUT_RDWR);
    close(fd);
}

static void nut_server_task(void *arg) {
    (void)arg;

    struct sockaddr_in addr = {0};
    addr.sin_family = AF_INET;
    addr.sin_port = htons(3493);
    addr.sin_addr.s_addr = htonl(INADDR_ANY);

    int fd = socket(AF_INET, SOCK_STREAM, IPPROTO_IP);
    if (fd < 0) {
        ESP_LOGE(TAG, "socket() failed");
        vTaskDelete(NULL);
        return;
    }

    int yes = 1;
    setsockopt(fd, SOL_SOCKET, SO_REUSEADDR, &yes, sizeof(yes));

    if (bind(fd, (struct sockaddr *)&addr, sizeof(addr)) != 0) {
        ESP_LOGE(TAG, "bind() failed");
        close(fd);
        vTaskDelete(NULL);
        return;
    }
    if (listen(fd, 2) != 0) {
        ESP_LOGE(TAG, "listen() failed");
        close(fd);
        vTaskDelete(NULL);
        return;
    }

    s_nut_listen_fd = fd;
    ESP_LOGI(TAG, "NUT server listening on tcp/3493");

    while (1) {
        struct sockaddr_in6 src_addr;
        socklen_t socklen = sizeof(src_addr);
        int cfd = accept(fd, (struct sockaddr *)&src_addr, &socklen);
        if (cfd < 0) continue;
        nut_handle_client(cfd);
    }
}

/* =========================
 * app_main
 * ========================= */
void app_main(void) {
    ESP_LOGI(TAG, "=======================================");
    ESP_LOGI(TAG, "Milestone v13 recovery fix (v13.1): AP-first config; no baked-in STA creds.");

    ESP_ERROR_CHECK(nvs_flash_init());
    cfg_load(&g_cfg);

    wifi_start_apsta(&g_cfg);
    start_http_server();

    // Start NUT server immediately (works on SoftAP and STA once connected).
    xTaskCreate(nut_server_task, "nutd", 6144, NULL, 5, NULL);

    ESP_LOGI(TAG, "Ready. Configure via SoftAP portal: http://192.168.4.1/config");
}
